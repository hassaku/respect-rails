def sanitize_params!
  if request_schema
    if validated.nil?
      validate_schema
    end
    if validated?
      request_schema.sanitize!(self)
    end
  end
end

def validate_schema
  log_msg = "  Request validation: "
  @validated = nil
  measure = Benchmark.realtime do
    @validated = request_schema.validate?(self) unless request_schema.nil?
  end
  if @validated.nil?
    log_msg += "none"
  else
    if @validated == true
      log_msg += "success"
    else
      log_msg += "failure"
    end
  end
  log_msg += " (%.1fms)" % [ measure * 1000 ]
  ::Rails.logger.info log_msg
  if @validated == false
    last_validation_error.context.each do |msg|
      ::Rails.logger.info "    #{msg}"
    end
    raise last_validation_error
  end
  true
end
